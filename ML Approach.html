from sklearn.model_selection import learning_curve
import matplotlib.pyplot as plt

def learningPlot(model, X = XTrain, y = yTrain, cv= None, title = ''):
    trainSize, trainScore, valid = learning_curve(model, XTrain, yTrain, cv = 12)
    plt.title(title)
    train_scores_mean = np.mean(trainScore, axis=1)
    train_scores_std = np.std(trainScore, axis=1)
    test_scores_mean = np.mean(valid, axis=1)
    test_scores_std = np.std(valid, axis=1)

    plt.grid()

    plt.fill_between(trainSize, train_scores_mean - train_scores_std,
                         train_scores_mean + train_scores_std, alpha=0.1,
                         color="r")
    plt.fill_between(trainSize, test_scores_mean - test_scores_std,
                         test_scores_mean + test_scores_std, alpha=0.1, color="g")
    plt.plot(trainSize, train_scores_mean, 'o-', color="r",
                 label="Training score")
    plt.plot(trainSize, test_scores_mean, 'o-', color="g",
                 label="Cross-validation score")

    plt.legend(loc="best")
    plt.show()
